Made with Visual Studio 2022 -> -std=c++11 and -std=c11 support (Visual Studio 2015, Visual Studio 2019 should work es well but Visual Studio 2017 not)
and cygwin with the following packges git diffutils nasm yasm pkgconf make. Important use host msys for x264

mv /usr/bin/link.exe /usr/bin/link.exe.bak
cd /cygdrive/c
git clone https://code.videolan.org/videolan/x264.git
cd /cygdrive/c/x264
curl "http://git.savannah.gnu.org/gitweb/?p=config.git;a=blob_plain;f=config.guess;hb=HEAD" > config.guess
find . -type f -print0 | xargs -0 dos2unix

CC=cl ./configure --prefix=installed --enable-static --host=msys --enable-debug --host=msys --extra-cflags="-MD -O2 -Z7"
CC=cl ./configure --prefix=installed --enable-static --host=msys --extra-cflags="-MD -O2 -Z7"

make
make install

cd ..
git clone https://git.ffmpeg.org/ffmpeg.git ffmpeg
cd ffmpeg
cp ../x264/installed installed

CC=cl PKG_CONFIG_PATH="installed/lib/pkgconfig" ./configure --prefix=installed --toolchain=msvc --target-os=win64 --arch=x86_64 --enable-asm --disable-shared --enable-static --enable-libx264 --enable-gpl --enable-nonfree --disable-doc --enable-debug --disable-optimizations --extra-ldflags="-LIBPATH:installed/lib  -LTCG:INCREMENTAL" --extra-cxxflags="-Iinstalled/include/ -MD -O1 -Z7" --extra-cflags="-Iinstalled/include/ -MD -O1 -Z7"
